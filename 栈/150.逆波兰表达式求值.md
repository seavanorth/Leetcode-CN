# 算法分析
遍历tokens的每个元素
- 若遇到数字，则转换为`int`后压栈
- 若遇到符号，则取出栈上方两个元素进行相应运算，并将结果压栈
- 最后返回栈顶
- 时间复杂度：O（n），n为tokens长度
- 空间复杂度：O（n），n为tokens长度
# 一、STL栈
```C++
class Solution
{
public:
    int evalRPN(vector<string> &tokens)
    {
        int len = tokens.size();
        stack<int> sta;
        for (int i = 0; i < len; i++)
        {
            string &token = tokens[i];
            if (token.length() > 1 || isdigit(token[0]))
                sta.push(atoi(token.c_str())); //将string转换成char*后再转换成int
            else
            {
                int n2 = sta.top(); //n2为右目运算数
                sta.pop();
                int n1 = sta.top(); //n1为左目运算数
                sta.pop();
                switch (token[0])
                {
                case '+':
                    sta.push(n1 + n2);
                    break;
                case '-':
                    sta.push(n1 - n2);
                    break;
                case '*':
                    sta.push(n1 * n2);
                    break;
                case '/':
                    sta.push(n1 / n2);
                    break;
                }
            }
        }
        return sta.top();
    }
};
```

# 数组模拟栈
```C++
class Solution
{
public:
    int evalRPN(vector<string> &tokens)
    {
        int len = tokens.size();
        vector<int> sta((len + 1) / 2);
        int top = -1;
        for (int i = 0; i < len; i++)
        {
            string &token = tokens[i];
            if (token.length() > 1 || isdigit(token[0]))
                sta[++top] = atoi(token.c_str()); //将string转换成char*后再转换成int
            else
            {
                int n2 = sta[top--]; //n2为右目运算数
                int n1 = sta[top--]; //n1为左目运算数
                switch (token[0])
                {
                case '+':
                    sta[++top] = n1 + n2;
                    break;
                case '-':
                    sta[++top] = n1 - n2;
                    break;
                case '*':
                    sta[++top] = n1 * n2;
                    break;
                case '/':
                    sta[++top] = n1 / n2;
                    break;
                }
            }
        }
        return sta[top];
    }
};
```
